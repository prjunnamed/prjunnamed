14#[derive(Debug)]
15pub enum Error {
16    Io(std::io::Error),
17    Json(jzon::Error),
18    Syntax(yosys::SyntaxError),
19    MetaDataType(yosys::MetadataTypeError),
20    Semantic,
21    Unsupported(String),
22}
23
24impl From<std::io::Error> for Error {
25    fn from(error: std::io::Error) -> Self {
26        Self::Io(error)
27    }
28}
29
30impl From<jzon::Error> for Error {
31    fn from(error: jzon::Error) -> Self {
32        Self::Json(error)
33    }
34}
35
36impl From<yosys::SyntaxError> for Error {
37    fn from(error: yosys::SyntaxError) -> Self {
38        Self::Syntax(error)
39    }
40}
41
42impl From<yosys::MetadataTypeError> for Error {
43    fn from(error: yosys::MetadataTypeError) -> Self {
44        Self::MetaDataType(error)
45    }
46}

48impl std::fmt::Display for Error {
49    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
50        match self {
51            Error::Io(error) => write!(f, "I/O error: {}", error),
52            Error::Json(error) => write!(f, "JSON parse error: {}", error),
53            Error::Syntax(error) => write!(f, "{}", error),
54            Error::MetaDataType(error) => write!(f, "{}", error),
55            Error::Semantic => write!(f, "semantic error"),
56            Error::Unsupported(feature) => write!(f, "unsupported feature: {}", feature),
57        }
58    }
59}

63struct ModuleImporter<'a> {
64    module: &'a yosys::Module,
65    design_io_ports: &'a BTreeSet<(&'a str, &'a str)>,
66    io_nets: BTreeMap<usize, IoNet>,
67    driven_nets: BTreeSet<usize>,
68    nets: BTreeMap<usize, Net>,
69    init: BTreeMap<usize, Trit>,
70    design: &'a Design,
71}

73impl ModuleImporter<'_> {
74    fn drive(&mut self, bits: &yosys::BitVector, value: impl Into<Value>) {
75        let value = value.into();
76        assert_eq!(bits.len(), value.len());
77        for (&bit, net) in bits.iter().zip(value.iter()) {
78            let yosys::Bit::Net(ynet) = bit else { unreachable!() };
79            assert!(!self.driven_nets.contains(&ynet));
80            match self.nets.entry(ynet) {
81                btree_map::Entry::Occupied(e) => {
82                    self.design.replace_net(*e.get(), net);
83                }
84                btree_map::Entry::Vacant(e) => {
85                    e.insert(net);
86                }
87            }
88            if let Some(&io_net) = self.io_nets.get(&ynet) {
89                self.design.add_iobuf(IoBuffer {
90                    enable: ControlNet::Pos(Net::ONE),
91                    output: Value::from(net),
92                    io: IoValue::from(io_net),
93                });
94            }
95            self.driven_nets.insert(ynet);
96        }
97    }
98
99    fn port_drive(&mut self, cell: &yosys::CellDetails, port: &str, value: impl Into<Value>) {
100        self.drive(cell.connections.get(port).unwrap(), value);
101    }
102
103    fn value(&mut self, bits: &yosys::BitVector) -> Value {
104        let mut nets = vec![];
105        for &bit in bits.iter() {
106            let net = match bit {
107                yosys::Bit::HiZ | yosys::Bit::Undef => Net::UNDEF,
108                yosys::Bit::Zero => Net::ZERO,
109                yosys::Bit::One => Net::ONE,
110                yosys::Bit::Net(ynet) => *self.nets.entry(ynet).or_insert_with(|| self.design.add_void(1).unwrap_net()),
111            };
112            nets.push(net);
113        }
114        Value::from_iter(nets)
115    }
116
117    fn io_value(&self, bits: &yosys::BitVector) -> IoValue {
118        let mut io = vec![];
119        for &bit in bits.iter() {
120            let yosys::Bit::Net(ynet) = bit else { unreachable!() };
121            let io_net = *self.io_nets.get(&ynet).unwrap();
122            io.push(io_net);
123        }
124        IoValue::from_iter(io)
125    }
126
127    fn port_value(&mut self, cell: &yosys::CellDetails, port: &str) -> Value {
128        self.value(cell.connections.get(port).unwrap())
129    }

131    fn port_control_net(&mut self, cell: &yosys::CellDetails, port: &str) -> ControlNet {
132        let net = self.port_value(cell, port).unwrap_net();
133        let polarity = cell.parameters.get(&format!("{port}_POLARITY")).unwrap().as_bool().unwrap();
134        if polarity {
135            ControlNet::Pos(net)
136        } else {
137            ControlNet::Neg(net)
138        }
139    }

141    fn init_value(&self, cell: &yosys::CellDetails, port: &str) -> Const {
142        let bits = cell.connections.get(port).unwrap();
143        let mut res = vec![];
144        for &bit in bits.iter() {
145            let yosys::Bit::Net(ynet) = bit else { unreachable!() };
146            res.push(match self.init.get(&ynet) {
147                None => Trit::Undef,
148                Some(&val) => val,
149            });
150        }
151        Const::from_iter(res)
152    }

154    fn value_ext(&mut self, cell: &yosys::CellDetails, port: &str, width: usize, signed: bool) -> Value {
155        if signed {
156            self.port_value(cell, port).sext(width)
157        } else {
158            self.port_value(cell, port).zext(width)
159        }
160    }

162    fn handle_init(&mut self) -> Result<(), Error> {
163        for net_details in self.module.netnames.0.values() {
164            let Some(init) = net_details.attributes.get("init") else { continue };
165            let init = init.as_const()?;
166            if init.len() != net_details.bits.len() {
167                Err(yosys::MetadataTypeError)?;
168            }
169            for (&bit, binit) in net_details.bits.iter().zip(init.iter()) {
170                if binit == Trit::Undef {
171                    continue;
172                }
173                let yosys::Bit::Net(ynet) = bit else { Err(yosys::MetadataTypeError)? };
174                match self.init.entry(ynet) {
175                    btree_map::Entry::Occupied(e) => {
176                        assert_eq!(*e.get(), binit);
177                    }
178                    btree_map::Entry::Vacant(e) => {
179                        e.insert(binit);
180                    }
181                }
182            }
183        }
184        Ok(())
185    }

187    fn handle_ports(&mut self) -> Result<(), Error> {
188        // determine set of nets that need to be IOs from cell port connections.
189        let mut io_net_conns = BTreeSet::new();
190        for cell in self.module.cells.0.values() {
191            for (port_name, bits) in cell.connections.iter() {
192                if self.design_io_ports.contains(&(&cell.type_, port_name)) {
193                    for &bit in bits.iter() {
194                        let yosys::Bit::Net(net) = bit else { return Err(Error::Semantic) };
195                        io_net_conns.insert(net);
196                    }
197                }
198            }
199        }
200
201        // now create all IoValues we're going to need.
202        let mut io_ports = BTreeSet::new();
203        for (port_name, port) in self.module.ports.iter() {
204            let mut is_inout = port.direction == yosys::PortDirection::Inout;
205            for &bit in port.bits.iter() {
206                if let yosys::Bit::Net(net) = bit {
207                    if io_net_conns.contains(&net) {
208                        is_inout = true;
209                    }
210                }
211            }
212            if is_inout {
213                let ioval = self.design.add_io(port_name.clone(), port.bits.0.len());
214                for (index, &bit) in port.bits.iter().enumerate() {
215                    let yosys::Bit::Net(net) = bit else { return Err(Error::Semantic) };
216                    match self.io_nets.entry(net) {
217                        btree_map::Entry::Occupied(_) => {
218                            return Err(Error::Semantic);
219                        }
220                        btree_map::Entry::Vacant(e) => {
221                            e.insert(ioval[index]);
222                        }
223                    }
224                }
225                io_ports.insert(port_name);
226            }
227        }
228        // handle all other ports
229        for (port_name, port) in self.module.ports.iter() {
230            if io_ports.contains(port_name) {
231                continue;
232            }
233            match port.direction {
234                yosys::PortDirection::Input => {
235                    let value = self.design.add_input(port_name, port.bits.len());
236                    self.drive(&port.bits, value);
237                }
238                yosys::PortDirection::Output => {
239                    let value = self.value(&port.bits);
240                    self.design.add_output(port_name, value);
241                }
242                yosys::PortDirection::Inout => unreachable!(),
243            }
244        }
245        Ok(())
246    }
247
248    fn handle_names(&mut self) -> Result<(), Error> {
249        for (name, details) in self.module.netnames.iter() {
250            if details.hide_name {
251                continue;
252            }
253            if details.bits.iter().any(|bit| {
254                if let yosys::Bit::Net(net) = bit {
255                    self.io_nets.contains_key(net)
256                } else {
257                    false
258                }
259            }) {
260                continue;
261            }
262            let value = self.value(&details.bits);
263            self.design.add_name(name, value);
264        }
265        Ok(())
266    }

268    fn use_attribute_metadata<'a>(design: &'a Design, attributes: &yosys::Metadata) -> WithMetadataGuard<'a> {
269        fn parse_position(text: &str) -> Option<SourcePosition> {
270            if let Some((line, column)) = text.split_once(".") {
271                Some(SourcePosition { line: line.parse::<u32>().ok()?, column: column.parse::<u32>().ok()? })
272            } else {
273                Some(SourcePosition { line: text.parse::<u32>().ok()?, column: 0 })
274            }
275        }
276
277        let mut items: Vec<MetaItemRef> = Vec::new();
278
279        // TODO: add a counter for failed src parsing, and emit warning if nonzero at the end of import
280        if let Some(yosys::MetadataValue::String(string)) = attributes.get("src") {
281            for piece in string.split("|") {
282                let Some((filename, loc_text)) = piece.rsplit_once(":") else {
283                    continue;
284                };
285                let (start_text, end_text) = loc_text.split_once("-").unwrap_or((loc_text, loc_text));
286                let Some(start) = parse_position(start_text) else {
287                    continue;
288                };
289                let Some(end) = parse_position(end_text) else {
290                    continue;
291                };
292
293                let meta_filename = design.add_metadata_string(filename);
294                items.push(design.add_metadata_item(&MetaItem::Source { file: meta_filename, start, end }));
295            }
296        }
297
298        design.use_metadata(MetaItemRef::from_iter(design, items))
299    }

301    fn handle_cell(&mut self, cell: &yosys::CellDetails) -> Result<(), Error> {
302        let _guard = ModuleImporter::use_attribute_metadata(&self.design, &cell.attributes);
303
304        match &cell.type_[..] {
305            "$not" | "$pos" | "$neg" => {
306                let width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
307                let a_signed = cell.parameters.get("A_SIGNED").unwrap().as_bool()?;
308                let a = self.value_ext(cell, "A", width, a_signed);
309                let value = match &cell.type_[..] {
310                    "$not" => self.design.add_not(a),
311                    "$pos" => a,
312                    "$neg" => {
313                        let value = self.design.add_not(a);
314                        self.design.add_adc(value, Value::zero(width), Net::ONE)[..width].into()
315                    }
316                    _ => unreachable!(),
317                };
318                self.port_drive(cell, "Y", value);
319            }
320            "$reduce_and" | "$reduce_or" | "$reduce_xor" | "$reduce_xnor" | "$reduce_bool" | "$logic_not" => {
321                let width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
322                let a = self.port_value(cell, "A");
323                let net = match &cell.type_[..] {
324                    "$reduce_and" => self.design.add_eq(Value::ones(a.len()), a),
325                    "$reduce_or" | "$reduce_bool" => self.design.add_ne(Value::zero(a.len()), a),
326                    "$reduce_xor" => {
327                        let mut net = Net::ZERO;
328                        for bit in &a {
329                            net = self.design.add_xor1(net, bit);
330                        }
331                        net
332                    }
333                    "$reduce_xnor" => {
334                        let mut net = Net::ONE;
335                        for bit in &a {
336                            net = self.design.add_xor1(net, bit);
337                        }
338                        net
339                    }
340                    "$logic_not" => self.design.add_eq(Value::zero(a.len()), a),
341                    _ => unreachable!(),
342                };
343                let mut value = Value::from(net);
344                if width == 0 {
345                    value = Value::new();
346                } else if width > 1 {
347                    value = value.zext(width);
348                }
349                self.port_drive(cell, "Y", value);
350            }
351            "$and" | "$or" | "$xor" | "$xnor" | "$add" | "$sub" | "$mul" | "$div" | "$mod" | "$divfloor"
352            | "$modfloor" => {
353                let width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
354                let a_signed = cell.parameters.get("A_SIGNED").unwrap().as_bool()?;
355                let b_signed = cell.parameters.get("B_SIGNED").unwrap().as_bool()?;
356                assert_eq!(a_signed, b_signed);
357                let a = self.value_ext(cell, "A", width, a_signed);
358                let b = self.value_ext(cell, "B", width, b_signed);
359                let value = match &cell.type_[..] {
360                    "$and" => self.design.add_and(a, b),
361                    "$or" => self.design.add_or(a, b),
362                    "$xor" => self.design.add_xor(a, b),
363                    "$xnor" => {
364                        let val = self.design.add_xor(a, b);
365                        self.design.add_not(val)
366                    }
367                    "$add" => self.design.add_adc(a, b, Net::ZERO)[..width].into(),
368                    "$sub" => {
369                        let inv_b = self.design.add_not(b);
370                        self.design.add_adc(a, inv_b, Net::ONE)[..width].into()
371                    }
372                    "$mul" => self.design.add_mul(a, b),
373                    "$div" | "$divfloor" if !a_signed => self.design.add_udiv(a, b),
374                    "$mod" | "$modfloor" if !a_signed => self.design.add_umod(a, b),
375                    "$div" => self.design.add_sdiv_trunc(a, b),
376                    "$mod" => self.design.add_smod_trunc(a, b),
377                    "$divfloor" => self.design.add_sdiv_floor(a, b),
378                    "$modfloor" => self.design.add_smod_floor(a, b),
379                    _ => unreachable!(),
380                };
381                self.port_drive(cell, "Y", value);
382            }
383            "$alu" => {
384                let width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
385                let bi = self.port_value(cell, "BI").unwrap_net();
386                let ci = self.port_value(cell, "CI").unwrap_net();
387                let a_signed = cell.parameters.get("A_SIGNED").unwrap().as_bool()?;
388                let b_signed = cell.parameters.get("B_SIGNED").unwrap().as_bool()?;
389                let a = self.value_ext(cell, "A", width, a_signed);
390                let b = self.value_ext(cell, "B", width, b_signed);
391                let b_inv = self.design.add_not(&b);
392                let b = self.design.add_mux(bi, b_inv, &b);
393                let x = self.design.add_xor(&a, &b);
394                self.port_drive(cell, "X", x);
395                let y = self.design.add_adc(&a, &b, ci);
396                self.port_drive(cell, "Y", &y[..width]);
397                let xor = self.design.add_xor(&a[1..], &b[1..]);
398                let co = self.design.add_xor(xor, &y[1..]).concat(y[width]);
399                self.port_drive(cell, "CO", co);
400            }
401            "$shl" | "$sshl" | "$shr" | "$sshr" | "$shift" | "$shiftx" => {
402                let width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
403                let a_signed = cell.parameters.get("A_SIGNED").unwrap().as_bool()?;
404                let b_signed = cell.parameters.get("B_SIGNED").unwrap().as_bool()?;
405                let mut a = self.port_value(cell, "A");
406                if a.len() < width {
407                    if cell.type_ == "$shiftx" {
408                        let a_width = a.len();
409                        a = a.concat(Value::undef(width - a_width))
410                    } else if a_signed {
411                        a = a.sext(width);
412                    } else {
413                        a = a.zext(width);
414                    }
415                }
416                let b = self.port_value(cell, "B");
417                let value = match &cell.type_[..] {
418                    "$shl" | "$sshl" => self.design.add_shl(a, b, 1),
419                    "$shr" => self.design.add_ushr(a, b, 1),
420                    "$sshr" => {
421                        if a_signed {
422                            self.design.add_sshr(a, b, 1)
423                        } else {
424                            self.design.add_ushr(a, b, 1)
425                        }
426                    }
427                    "$shift" | "$shiftx" => {
428                        let val_shr = if cell.type_ == "$shiftx" {
429                            self.design.add_xshr(&a, &b, 1)
430                        } else if a_signed {
431                            self.design.add_sshr(&a, &b, 1)
432                        } else {
433                            self.design.add_ushr(&a, &b, 1)
434                        };
435                        if b_signed {
436                            let b_inv = self.design.add_not(&b);
437                            let b_neg =
438                                Value::from(&self.design.add_adc(Value::zero(b.len()), &b_inv, Net::ONE)[..b.len()]);
439                            let val_shl = self.design.add_shl(&a, b_neg, 1);
440                            self.design.add_mux(b[b.len() - 1], val_shl, val_shr)
441                        } else {
442                            val_shr
443                        }
444                    }
445                    _ => unreachable!(),
446                };
447                self.port_drive(cell, "Y", &value[..width]);
448            }
449            "$lt" | "$le" | "$gt" | "$ge" | "$eq" | "$ne" => {
450                let y_width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
451                let a_width = cell.parameters.get("A_WIDTH").unwrap().as_i32()? as usize;
452                let b_width = cell.parameters.get("B_WIDTH").unwrap().as_i32()? as usize;
453                let a_signed = cell.parameters.get("A_SIGNED").unwrap().as_bool()?;
454                let b_signed = cell.parameters.get("B_SIGNED").unwrap().as_bool()?;
455                let width = std::cmp::max(a_width, b_width);
456                assert_eq!(a_signed, b_signed);
457                let a = self.value_ext(cell, "A", width, a_signed);
458                let b = self.value_ext(cell, "B", width, b_signed);
459                let (mut net, inv) = match &cell.type_[..] {
460                    "$lt" if !a_signed => (self.design.add_ult(a, b), false),
461                    "$gt" if !a_signed => (self.design.add_ult(b, a), false),
462                    "$le" if !a_signed => (self.design.add_ult(b, a), true),
463                    "$ge" if !a_signed => (self.design.add_ult(a, b), true),
464                    "$lt" if a_signed => (self.design.add_slt(a, b), false),
465                    "$gt" if a_signed => (self.design.add_slt(b, a), false),
466                    "$le" if a_signed => (self.design.add_slt(b, a), true),
467                    "$ge" if a_signed => (self.design.add_slt(a, b), true),
468                    "$eq" => (self.design.add_eq(a, b), false),
469                    "$ne" => (self.design.add_eq(a, b), true),
470                    _ => unreachable!(),
471                };
472                if inv {
473                    net = self.design.add_not1(net);
474                }
475                let mut value = Value::from(net);
476                if y_width == 0 {
477                    value = Value::new();
478                } else if y_width > 1 {
479                    value = value.zext(width);
480                }
481                self.port_drive(cell, "Y", value);
482            }
483            "$logic_and" | "$logic_or" => {
484                let y_width = cell.parameters.get("Y_WIDTH").unwrap().as_i32()? as usize;
485                let a = self.port_value(cell, "A");
486                let a = self.design.add_eq(Value::zero(a.len()), a);
487                let a = self.design.add_not(a);
488                let b = self.port_value(cell, "B");
489                let b = self.design.add_eq(Value::zero(b.len()), b);
490                let b = self.design.add_not(b);
491                let mut value = match &cell.type_[..] {
492                    "$logic_and" => self.design.add_and(a, b),
493                    "$logic_or" => self.design.add_or(a, b),
494                    _ => unreachable!(),
495                };
496                if y_width == 0 {
497                    value = Value::new();
498                } else if y_width > 1 {
499                    value = value.zext(y_width);
500                }
501                self.port_drive(cell, "Y", value);
502            }
503            "$mux" => {
504                let a = self.port_value(cell, "A");
505                let b = self.port_value(cell, "B");
506                let s = self.port_value(cell, "S");
507                assert_eq!(a.len(), b.len());
508                let y = self.design.add_mux(s.unwrap_net(), b, a);
509                self.port_drive(cell, "Y", y);
510            }
511            "$bmux" => {
512                let mut value = self.port_value(cell, "A");
513                let sel = self.port_value(cell, "S");
514                for s in sel.iter().rev() {
515                    assert_eq!(value.len() % 2, 0);
516                    let lo = Value::from(&value[..(value.len() / 2)]);
517                    let hi = Value::from(&value[(value.len() / 2)..]);
518                    value = self.design.add_mux(s, hi, lo);
519                }
520                self.port_drive(cell, "Y", value);
521            }
522            "$pmux" => {
523                let mut value = self.port_value(cell, "A");
524                let b = self.port_value(cell, "B");
525                let sel = self.port_value(cell, "S");
526                assert_eq!(b.len(), value.len() * sel.len());
527                for (i, s) in sel.iter().enumerate() {
528                    value = self.design.add_mux(s, &b[(i * value.len())..((i + 1) * value.len())], value);
529                }
530                self.port_drive(cell, "Y", value);
531            }
532            "$tribuf" => {
533                let output = self.port_value(cell, "A");
534                let enable = self.port_value(cell, "EN");
535                let enable = ControlNet::Pos(enable.unwrap_net());
536                let bits = cell.connections.get("Y").unwrap();
537                let io = self.io_value(bits);
538                let value = self.design.add_iobuf(IoBuffer { output, enable, io });
539                for (&bit, net) in bits.iter().zip(value.iter()) {
540                    let yosys::Bit::Net(ynet) = bit else { unreachable!() };
541                    assert!(!self.driven_nets.contains(&ynet));
542                    match self.nets.entry(ynet) {
543                        btree_map::Entry::Occupied(e) => {
544                            self.design.replace_net(*e.get(), net);
545                        }
546                        btree_map::Entry::Vacant(e) => {
547                            e.insert(net);
548                        }
549                    }
550                    self.driven_nets.insert(ynet);
551                }
552            }
553            "$dff" | "$dffe" | "$adff" | "$adffe" | "$sdff" | "$sdffe" | "$sdffce" => {
554                let data = self.port_value(cell, "D");
555                let enable = if cell.connections.contains_key("EN") {
556                    self.port_control_net(cell, "EN")
557                } else {
558                    ControlNet::Pos(Net::ONE)
559                };
560                let (clear, clear_value) = if cell.connections.contains_key("ARST") {
561                    (self.port_control_net(cell, "ARST"), cell.parameters.get("ARST_VALUE").unwrap().as_const()?)
562                } else {
563                    (ControlNet::Pos(Net::ZERO), Const::undef(data.len()))
564                };
565                let (reset, reset_value) = if cell.connections.contains_key("SRST") {
566                    (self.port_control_net(cell, "SRST"), cell.parameters.get("SRST_VALUE").unwrap().as_const()?)
567                } else {
568                    (ControlNet::Pos(Net::ZERO), Const::undef(data.len()))
569                };
570                let clock = self.port_control_net(cell, "CLK");
571                let init_value = self.init_value(cell, "Q");
572                let q = self.design.add_dff(FlipFlop {
573                    data,
574                    clock,
575                    enable,
576                    reset,
577                    reset_over_enable: cell.type_ != "$sdffce",
578                    clear,
579                    init_value,
580                    reset_value,
581                    clear_value,
582                });
583                self.port_drive(cell, "Q", q);
584            }
585            "$mem_v2" => {
586                let offset = usize::try_from(cell.parameters.get("OFFSET").unwrap().as_i32()?).unwrap();
587                let size = usize::try_from(cell.parameters.get("SIZE").unwrap().as_i32()?).unwrap();
588                let depth = offset + size;
589                let abits = usize::try_from(cell.parameters.get("ABITS").unwrap().as_i32()?).unwrap();
590                let width = usize::try_from(cell.parameters.get("WIDTH").unwrap().as_i32()?).unwrap();
591                let mut init_value = cell.parameters.get("INIT").unwrap().as_const().unwrap();
592                assert_eq!(init_value.len(), size * width);
593                init_value = Const::undef(offset * width).concat(init_value);
594
595                let mut memory = Memory { depth, width, init_value, write_ports: vec![], read_ports: vec![] };
596
597                let wr_ports = usize::try_from(cell.parameters.get("WR_PORTS").unwrap().as_i32()?).unwrap();
598                let wr_clk_enable = cell.parameters.get("WR_CLK_ENABLE").unwrap().as_const().unwrap();
599                let wr_clk_polarity = cell.parameters.get("WR_CLK_POLARITY").unwrap().as_const().unwrap();
600                let wr_priority_mask = cell.parameters.get("WR_PRIORITY_MASK").unwrap().as_const().unwrap();
601                let wr_wide_continuation = cell.parameters.get("WR_WIDE_CONTINUATION").unwrap().as_const().unwrap();
602                let wr_clk = self.port_value(cell, "WR_CLK");
603                let wr_en = self.port_value(cell, "WR_EN");
604                let wr_addr = self.port_value(cell, "WR_ADDR");
605                let wr_data = self.port_value(cell, "WR_DATA");
606                let mut wr_port_indices = vec![];
607                if wr_ports != 0 {
608                    assert!(wr_priority_mask.iter().all(|x| x == Trit::Zero));
609                    assert!(wr_clk_enable.iter().all(|x| x == Trit::One));
610                }
611                let mut index = 0;
612                while index < wr_ports {
613                    let mut end_index = index + 1;
614                    while end_index < wr_ports && wr_wide_continuation[end_index] == Trit::One {
615                        end_index += 1;
616                    }
617                    let wide = end_index - index;
618                    let wide_log2 = wide.ilog2() as usize;
619                    assert!(wide.is_power_of_two());
620                    wr_port_indices.push(index);
621                    let clock = if wr_clk_polarity[index] == Trit::One {
622                        ControlNet::Pos(wr_clk[index])
623                    } else {
624                        ControlNet::Neg(wr_clk[index])
625                    };
626                    let addr = wr_addr.slice(index * abits..(index + 1) * abits).slice(wide_log2..);
627                    let data = wr_data.slice(index * width..end_index * width);
628                    let mask = wr_en.slice(index * width..end_index * width);
629                    memory.write_ports.push(MemoryWritePort { clock, addr, data, mask });
630                    index = end_index;
631                }
632
633                let rd_ports = usize::try_from(cell.parameters.get("RD_PORTS").unwrap().as_i32()?).unwrap();
634                let rd_clk_enable = cell.parameters.get("RD_CLK_ENABLE").unwrap().as_const().unwrap();
635                let rd_clk_polarity = cell.parameters.get("RD_CLK_POLARITY").unwrap().as_const().unwrap();
636                let rd_transparency_mask = cell.parameters.get("RD_TRANSPARENCY_MASK").unwrap().as_const().unwrap();
637                let rd_collision_x_mask = cell.parameters.get("RD_COLLISION_X_MASK").unwrap().as_const().unwrap();
638                let rd_wide_continuation = cell.parameters.get("RD_WIDE_CONTINUATION").unwrap().as_const().unwrap();
639                let rd_ce_over_srst = cell.parameters.get("RD_CE_OVER_SRST").unwrap().as_const().unwrap();
640                let rd_arst_value = cell.parameters.get("RD_ARST_VALUE").unwrap().as_const().unwrap();
641                let rd_srst_value = cell.parameters.get("RD_SRST_VALUE").unwrap().as_const().unwrap();
642                let rd_init_value = cell.parameters.get("RD_INIT_VALUE").unwrap().as_const().unwrap();
643                let rd_clk = self.port_value(cell, "RD_CLK");
644                let rd_en = self.port_value(cell, "RD_EN");
645                let rd_srst = self.port_value(cell, "RD_SRST");
646                let rd_arst = self.port_value(cell, "RD_ARST");
647                let rd_addr = self.port_value(cell, "RD_ADDR");
648                let mut index = 0;
649                while index < rd_ports {
650                    let mut end_index = index + 1;
651                    while end_index < wr_ports && rd_wide_continuation[end_index] == Trit::One {
652                        end_index += 1;
653                    }
654                    let wide = end_index - index;
655                    let wide_log2 = wide.ilog2() as usize;
656                    assert!(wide.is_power_of_two());
657                    let flip_flop = if rd_clk_enable[index] == Trit::One {
658                        let clock = if rd_clk_polarity[index] == Trit::One {
659                            ControlNet::Pos(rd_clk[index])
660                        } else {
661                            ControlNet::Neg(rd_clk[index])
662                        };
663                        let mut relations = vec![];
664                        for (new_index, &orig_index) in wr_port_indices.iter().enumerate() {
665                            let mask_index = index * wr_ports + orig_index;
666                            let relation = if memory.write_ports[new_index].clock != clock
667                                || rd_collision_x_mask[mask_index] == Trit::One
668                            {
669                                MemoryPortRelation::Undefined
670                            } else if rd_transparency_mask[mask_index] == Trit::One {
671                                MemoryPortRelation::Transparent
672                            } else {
673                                MemoryPortRelation::ReadBeforeWrite
674                            };
675                            relations.push(relation);
676                        }
677                        Some(MemoryReadFlipFlop {
678                            clock,
679                            clear: rd_arst[index].into(),
680                            reset: rd_srst[index].into(),
681                            enable: rd_en[index].into(),
682                            reset_over_enable: rd_ce_over_srst[index] != Trit::One,
683                            clear_value: rd_arst_value.slice(index * width..end_index * width),
684                            reset_value: rd_srst_value.slice(index * width..end_index * width),
685                            init_value: rd_init_value.slice(index * width..end_index * width),
686                            relations,
687                        })
688                    } else {
689                        None
690                    };
691                    let addr = rd_addr.slice(index * abits..(index + 1) * abits).slice(wide_log2..);
692                    let data_len = wide * width;
693                    memory.read_ports.push(MemoryReadPort { addr, data_len, flip_flop });
694                    index = end_index;
695                }
696
697                let mem_out = self.design.add_memory(memory);
698                self.port_drive(cell, "RD_DATA", mem_out);
699            }
700            "$scopeinfo" => {
701                // not quite yet
702            }
703            "$memrd" | "$memwr" | "$meminit" | "$memrd_v2" | "$memwr_v2" | "$meminit_v2" => {
704                return Err(Error::Unsupported(format!(
705                    "{} cell; run the Yosys `memory_collect` pass before writing JSON",
706                    cell.type_
707                )))
708            }
709            _ => {
710                if cell.type_.starts_with('$') {
711                    return Err(Error::Unsupported(format!("{} cell", cell.type_)));
712                }
713                // instance
714                let mut out_bits = vec![];
715                let mut next_out = 0;
716                let mut inputs = BTreeMap::new();
717                let mut outputs = BTreeMap::new();
718                let mut ios = BTreeMap::new();
719                for (name, bits) in cell.connections.iter() {
720                    let direction = *cell.port_directions.get(name).unwrap();
721                    if self.design_io_ports.contains(&(&cell.type_, name)) || direction == yosys::PortDirection::Inout {
722                        ios.insert(name.clone(), self.io_value(bits));
723                    } else if direction == yosys::PortDirection::Output {
724                        let range = next_out..(next_out + bits.len());
725                        next_out += bits.len();
726                        out_bits.push((bits, range.clone()));
727                        outputs.insert(name.clone(), range);
728                    } else {
729                        inputs.insert(name.clone(), self.value(bits));
730                    }
731                }
732                let mut parameters = BTreeMap::new();
733                for (name, val) in cell.parameters.iter() {
734                    let val = match val {
735                        yosys::MetadataValue::Const(val) => ParamValue::Const(val.clone()),
736                        yosys::MetadataValue::String(val) => ParamValue::String(val.clone()),
737                    };
738                    parameters.insert(name.clone(), val);
739                }
740                let output = self.design.add_other(Instance {
741                    kind: cell.type_.strip_prefix('\\').unwrap_or(cell.type_.as_str()).to_string(),
742                    params: parameters,
743                    inputs,
744                    outputs,
745                    ios,
746                });
747                for (bits, range) in out_bits {
748                    self.drive(bits, &output[range]);
749                }
750            }
751        }
752        Ok(())
753    }
754
755    fn handle_undriven_nets(&mut self) {
756        for (&ynet, &io_net) in &self.io_nets {
757            if self.driven_nets.contains(&ynet) {
758                continue;
759            }
760            let Some(&net) = self.nets.get(&ynet) else { continue };
761            self.design.replace_net(
762                net,
763                self.design
764                    .add_iobuf(IoBuffer {
765                        output: Value::undef(1),
766                        enable: ControlNet::Pos(Net::ZERO),
767                        io: io_net.into(),
768                    })
769                    .unwrap_net(),
770            );
771            self.driven_nets.insert(ynet);
772        }
773        for (&ynet, &net) in &self.nets {
774            if self.driven_nets.contains(&ynet) {
775                continue;
776            }
777            self.design.replace_net(net, Net::UNDEF);
778        }
779    }
780}

782fn import_module(
783    target: Option<Arc<dyn Target>>,
784    module: &yosys::Module,
785    design_io_ports: &BTreeSet<(&str, &str)>,
786) -> Result<Option<Design>, Error> {
787    if let Some(val) = module.attributes.get("blackbox") {
788        if val.as_bool()? {
789            return Ok(None);
790        }
791    }
792
793    let mut design = Design::with_target(target);
794
795    let mut importer = ModuleImporter {
796        module,
797        design_io_ports,
798        io_nets: BTreeMap::new(),
799        driven_nets: BTreeSet::new(),
800        nets: BTreeMap::new(),
801        init: BTreeMap::new(),
802        design: &design,
803    };
804
805    importer.handle_init()?;
806    importer.handle_ports()?;
807    importer.handle_names()?;
808    for cell in module.cells.0.values() {
809        importer.handle_cell(cell)?;
810    }
811    importer.handle_undriven_nets();
812    design.compact();
813
814    Ok(Some(design))
815}
816
817fn index_io_ports(design: &yosys::Design) -> Result<BTreeSet<(&str, &str)>, Error> {
818    let mut io_ports: BTreeSet<(&str, &str)> = BTreeSet::new();
819    io_ports.insert(("$tribuf", "Y"));
820    for (mod_name, module) in design.modules.iter() {
821        for (port_name, port) in module.ports.iter() {
822            if port.direction == yosys::PortDirection::Inout {
823                io_ports.insert((mod_name, port_name));
824                continue;
825            }
826            let Some(net_details) = module.netnames.get(port_name) else { continue };
827            if let Some(val) = net_details.attributes.get("iopad_external_pin") {
828                if val.as_bool()? == true {
829                    io_ports.insert((mod_name, port_name));
830                }
831            }
832        }
833    }
834    Ok(io_ports)
835}
836
837pub fn import(
838    target: Option<Arc<dyn Target>>,
839    reader: &mut impl std::io::Read,
840) -> Result<BTreeMap<String, Design>, Error> {
841    let mut text = String::new();
842    reader.read_to_string(&mut text)?;
843    let json = jzon::parse(text.as_str())?;
844    let yosys_design = yosys::Design::try_from(json)?;
845
846    let io_ports = index_io_ports(&yosys_design)?;
847    let mut designs = BTreeMap::new();
848    for (name, module) in yosys_design.modules.iter() {
849        if let Some(design) = import_module(target.clone(), module, &io_ports)? {
850            designs.insert(name.clone(), design);
851        }
852    }
853    Ok(designs)
854}
