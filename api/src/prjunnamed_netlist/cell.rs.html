19/// Compact, 16-byte representation for cells common in fine netlists.
20///
21/// This type is used internally within [`Design`]. Cells that can be
22/// represented with [`CellRepr`] get stored without a separate heap
23/// allocation. All others get represented as [`CellRepr::Coarse`].
24#[derive(Debug, Clone, PartialEq, Eq, Hash)]
25pub(crate) enum CellRepr {
26    /// Reserved index. Can be created with [`Design::add_void`] for later
27    /// replacement with [`Design::replace_value`], or left over
28    /// after [`CellRef::unalive`].
29    ///
30    /// [`CellRef::unalive`]: crate::CellRef::unalive
31    Void,
32
33    /// Represents an index taken by a cell with more than one bit of
34    /// output. Stores the base index of the cell in question.
35    Skip(u32),
36
37    Buf(Net),
38    Not(Net),
39    And(Net, Net),
40    Or(Net, Net),
41    Xor(Net, Net),
42    Mux(Net, Net, Net), // a ? b : c
43    Adc(Net, Net, Net), // a + b + ci
44
45    Boxed(Box<Cell>),
46}
47
48/// A unit of logic.
49///
50/// Within a [`Design`], each cell is identified by a range of indices,
51/// with each index corresponding to one of the output bits of the cell.
52/// As such, the cell itself only contains information about its inputs,
53/// with the outputs being implicit.
54///
55/// (do note that cells without any outputs, such as [`output`][Cell::Output]
56/// or [`name`][Cell::Name], still get assigned an index)
57#[derive(Clone, Debug, PartialEq, Eq, Hash)]
58pub enum Cell {
59    Buf(Value),
60    Not(Value),
61    /// `a & b`.
62    ///
63    /// Has short-circuiting behavior for inputs containing `X` — if the other
64    /// bit is `0`, the output is `0` and the `X` doesn't propagate.
65    And(Value, Value),
66    /// `a | b`.
67    ///
68    /// Has short-circuiting behavior for inputs containing `X` — if the other
69    /// bit is `1`, the output is `1` and the `X` doesn't propagate.
70    Or(Value, Value),
71    Xor(Value, Value),
72    /// `a ? b : c`.
73    ///
74    /// Muxes are glitch free — if `a` is `X`, the bit positions that match
75    /// between `b` and `c` still have a defined value. The `X` propagates
76    /// only at the positions where `b` and `c` differ.
77    Mux(Net, Value, Value),
78    /// `a + b + ci` — add with carry.
79    ///
80    /// Output is one bit wider than `a` and `b` — the most significant bit
81    /// is the carry-out.
82    ///
83    /// `X`s in the input propagate only to the more significant bits, and
84    /// do not affect the less significant bits.
85    Adc(Value, Value, Net), // a + b + ci
86
87    Eq(Value, Value),
88    ULt(Value, Value),
89    SLt(Value, Value),
90
91    /// `a << (b * c)`. The bottom bits are filled with zeros.
92    ///
93    /// General notes for all shift cells:
94    /// - output is the same width as `a`. If you need wider output,
95    ///   zero-extend or sign-extend your input first, as appropriate.
96    /// - the shift count does not wrap. If you shift by more than
97    ///   `a.len() - 1`, you get the same result as if you made an equivalent
98    ///   sequence of 1-bit shifts (i.e. all zeros, all sign bits, or all `X`,
99    ///   as appropriate).
100    /// - shift cells are one of the few cells which *do not* expect their
101    ///   inputs to be of the same width. In fact, that is the expected case.
102    Shl(Value, Value, u32),
103    /// `a >> (b * c)`. The top bits are filled with zeros.
104    ///
105    /// See also [general notes above][Cell::Shl].
106    UShr(Value, Value, u32),
107    /// `a >> (b * c)`. The top bits are filled with copies of the top bit
108    /// of the input.
109    ///
110    /// `a` must be at least one bit wide (as otherwise there would be no sign
111    /// bit to propagate, and while there wouldn't be anywhere to propagate it
112    /// *to*, it's an edge-case it doesn't make sense to bother handling).
113    ///
114    /// See also [general notes above][Cell::Shl].
115    SShr(Value, Value, u32),
116    /// `a >> (b * c)`. The top bits are filled with `X`.
117    ///
118    /// See also [general notes above][Cell::Shl].
119    XShr(Value, Value, u32),
120
121    // future possibilities: popcnt, count leading/trailing zeros, powers
122    Mul(Value, Value),
123    UDiv(Value, Value),
124    UMod(Value, Value),
125    SDivTrunc(Value, Value),
126    SDivFloor(Value, Value),
127    SModTrunc(Value, Value),
128    SModFloor(Value, Value),
129
130    Match(MatchCell),
131    Assign(AssignCell),
132
133    Dff(FlipFlop),
134    Memory(Memory),
135    IoBuf(IoBuffer),
136    Target(TargetCell),
137    Other(Instance),
138
139    /// Design input of a given width.
140    ///
141    /// If synthesizing for a specified target, and not in out-of-context mode,
142    /// an input will be replaced with an [`IoBuffer`] and attached to a pin on
143    /// the target device.
144    Input(String, usize),
145    /// Design output. Attaches a name to a given value.
146    ///
147    /// If synthesizing for a specified target, and not in out-of-context mode,
148    /// an output will be replaced with an [`IoBuffer`] and attached to a pin on
149    /// the target device.
150    Output(String, Value),
151    /// Attaches a name to a given value for debugging.
152    ///
153    /// `Name` keeps a given value alive during optimization and makes it easily
154    /// available to be poked at during simulation.
155    ///
156    /// Do note that the [`unname` pass][unname], which runs during
157    /// target-dependent synthesis, replaces all `Name` cells with [`Debug`]
158    /// cells.
159    ///
160    /// [unname]: ../prjunnamed_generic/fn.unname.html
161    /// [`Debug`]: Cell::Debug
162    Name(String, Value),
163    /// Tentatively attaches a name to a given value.
164    ///
165    /// `Debug` gives a name to a particular value, without insisting on keeping
166    /// it alive during optimization. This helps correlate the output of
167    /// synthesis with the corresponding input logic.
168    ///
169    /// If at any point a value is being kept alive only by a `Debug` cell,
170    /// it will be optimized out and the input to the `Debug` cell will
171    /// be replaced with `X`.
172    ///
173    /// See also: [`Name`][Cell::Name].
174    Debug(String, Value),
175}

177impl Cell {
178    pub fn validate(&self, design: &Design) {
179        match self {
180            Cell::Buf(_) => (),
181            Cell::Not(_) => (),
182            Cell::And(arg1, arg2)
183            | Cell::Or(arg1, arg2)
184            | Cell::Xor(arg1, arg2)
185            | Cell::Mux(_, arg1, arg2)
186            | Cell::Adc(arg1, arg2, _)
187            | Cell::Eq(arg1, arg2)
188            | Cell::ULt(arg1, arg2)
189            | Cell::Mul(arg1, arg2)
190            | Cell::UDiv(arg1, arg2)
191            | Cell::UMod(arg1, arg2) => assert_eq!(arg1.len(), arg2.len()),
192            Cell::SLt(arg1, arg2)
193            | Cell::SDivTrunc(arg1, arg2)
194            | Cell::SDivFloor(arg1, arg2)
195            | Cell::SModTrunc(arg1, arg2)
196            | Cell::SModFloor(arg1, arg2) => {
197                assert_eq!(arg1.len(), arg2.len());
198                assert!(!arg1.is_empty());
199            }
200
201            Cell::Shl(..) => (),
202            Cell::UShr(..) => (),
203            Cell::SShr(arg1, _, _) => assert!(!arg1.is_empty()),
204            Cell::XShr(..) => (),
205
206            Cell::Match(match_cell) => {
207                for alternates in &match_cell.patterns {
208                    for pattern in alternates {
209                        assert_eq!(match_cell.value.len(), pattern.len());
210                    }
211                }
212            }
213            Cell::Assign(assign_cell) => {
214                assert!(assign_cell.value.len() >= assign_cell.update.len() + assign_cell.offset);
215            }
216
217            Cell::Dff(flip_flop) => {
218                assert_eq!(flip_flop.data.len(), flip_flop.init_value.len());
219                assert_eq!(flip_flop.data.len(), flip_flop.clear_value.len());
220                assert_eq!(flip_flop.data.len(), flip_flop.reset_value.len());
221            }
222            Cell::Memory(memory) => {
223                assert_eq!(memory.init_value.len(), memory.depth * memory.width);
224                for port in &memory.write_ports {
225                    assert_eq!(port.data.len(), port.mask.len());
226                    if memory.width == 0 {
227                        assert_eq!(port.data.len(), 0);
228                    } else {
229                        assert_eq!(port.data.len() % memory.width, 0);
230                        let wide_factor = port.data.len() / memory.width;
231                        assert!(wide_factor.is_power_of_two());
232                        assert_eq!(memory.depth % wide_factor, 0);
233                    }
234                }
235                for port in &memory.read_ports {
236                    if memory.width == 0 {
237                        assert_eq!(port.data_len, 0);
238                    } else {
239                        assert_eq!(port.data_len % memory.width, 0);
240                        let wide_factor = port.data_len / memory.width;
241                        assert!(wide_factor.is_power_of_two());
242                        assert_eq!(memory.depth % wide_factor, 0);
243                    }
244                    if let Some(ref flip_flop) = port.flip_flop {
245                        assert_eq!(flip_flop.clear_value.len(), port.data_len);
246                        assert_eq!(flip_flop.reset_value.len(), port.data_len);
247                        assert_eq!(flip_flop.init_value.len(), port.data_len);
248                        assert_eq!(flip_flop.relations.len(), memory.write_ports.len());
249                        for (write_port_index, &relation) in flip_flop.relations.iter().enumerate() {
250                            if relation != MemoryPortRelation::Undefined {
251                                assert_eq!(memory.write_ports[write_port_index].clock, flip_flop.clock);
252                            }
253                        }
254                    }
255                }
256            }
257            Cell::IoBuf(io_buffer) => {
258                assert_eq!(io_buffer.output.len(), io_buffer.io.len());
259            }
260            Cell::Target(target_cell) => {
261                let prototype = design.target_prototype(target_cell);
262                assert_eq!(target_cell.params.len(), prototype.params.len());
263                for (param, value) in prototype.params.iter().zip(target_cell.params.iter()) {
264                    assert!(param.kind.is_valid(value));
265                }
266                assert_eq!(target_cell.inputs.len(), prototype.input_len);
267                assert_eq!(target_cell.output_len, prototype.output_len);
268                assert_eq!(target_cell.ios.len(), prototype.io_len);
269                let target = design.target().unwrap();
270                target.validate(design, target_cell);
271            }
272            Cell::Other(_instance) => {
273                // TODO
274            }
275            Cell::Input(..) => (),
276            Cell::Output(..) => (),
277            Cell::Name(..) | Cell::Debug(..) => (),
278        }
279    }
280
281    /// If possible, return a cell that computes only a slice of the outputs
282    /// of this cell.
283    pub fn slice(&self, range: impl std::ops::RangeBounds<usize> + Clone) -> Option<Cell> {
284        match self {
285            Cell::Buf(arg) => Some(Cell::Buf(arg.slice(range))),
286            Cell::Not(arg) => Some(Cell::Not(arg.slice(range))),
287            Cell::And(arg1, arg2) => Some(Cell::And(arg1.slice(range.clone()), arg2.slice(range))),
288            Cell::Or(arg1, arg2) => Some(Cell::Or(arg1.slice(range.clone()), arg2.slice(range))),
289            Cell::Xor(arg1, arg2) => Some(Cell::Xor(arg1.slice(range.clone()), arg2.slice(range))),
290            Cell::Mux(arg1, arg2, arg3) => Some(Cell::Mux(*arg1, arg2.slice(range.clone()), arg3.slice(range))),
291            Cell::Dff(flip_flop) => Some(Cell::Dff(flip_flop.slice(range))),
292            Cell::IoBuf(io_buffer) => Some(Cell::IoBuf(io_buffer.slice(range))),
293            _ => None,
294        }
295    }

310impl CellRepr {
311    pub fn get(&self) -> Cow<'_, Cell> {
312        match *self {
313            CellRepr::Void => unreachable!("void cell"),
314            CellRepr::Skip(_) => unreachable!("skip cell"),
315
316            CellRepr::Buf(arg) => Cow::Owned(Cell::Buf(Value::from(arg))),
317            CellRepr::Not(arg) => Cow::Owned(Cell::Not(Value::from(arg))),
318            CellRepr::And(arg1, arg2) => Cow::Owned(Cell::And(Value::from(arg1), Value::from(arg2))),
319            CellRepr::Or(arg1, arg2) => Cow::Owned(Cell::Or(Value::from(arg1), Value::from(arg2))),
320            CellRepr::Xor(arg1, arg2) => Cow::Owned(Cell::Xor(Value::from(arg1), Value::from(arg2))),
321            CellRepr::Mux(arg1, arg2, arg3) => Cow::Owned(Cell::Mux(arg1, Value::from(arg2), Value::from(arg3))),
322            CellRepr::Adc(arg1, arg2, arg3) => Cow::Owned(Cell::Adc(Value::from(arg1), Value::from(arg2), arg3)),
323
324            CellRepr::Boxed(ref cell) => Cow::Borrowed(cell),
325        }
326    }
327}

329impl From<Cell> for CellRepr {
330    fn from(value: Cell) -> Self {
331        match value {
332            Cell::Buf(arg) if arg.len() == 1 => CellRepr::Buf(arg[0]),
333            Cell::Not(arg) if arg.len() == 1 => CellRepr::Not(arg[0]),
334            Cell::And(arg1, arg2) if arg1.len() == 1 && arg2.len() == 1 => CellRepr::And(arg1[0], arg2[0]),
335            Cell::Or(arg1, arg2) if arg1.len() == 1 && arg2.len() == 1 => CellRepr::Or(arg1[0], arg2[0]),
336            Cell::Xor(arg1, arg2) if arg1.len() == 1 && arg2.len() == 1 => CellRepr::Xor(arg1[0], arg2[0]),
337            Cell::Mux(arg1, arg2, arg3) if arg2.len() == 1 && arg3.len() == 1 => CellRepr::Mux(arg1, arg2[0], arg3[0]),
338            Cell::Adc(arg1, arg2, arg3) if arg1.len() == 1 && arg2.len() == 1 => CellRepr::Adc(arg1[0], arg2[0], arg3),
339
340            cell => CellRepr::Boxed(Box::new(cell)),
341        }
342    }
343}

345impl CellRepr {
346    pub fn output_len(&self) -> usize {
347        match self {
348            CellRepr::Void => unreachable!("void cell"),
349            CellRepr::Skip(_) => unreachable!("skip cell"),
350
351            CellRepr::Buf(..)
352            | CellRepr::Not(..)
353            | CellRepr::And(..)
354            | CellRepr::Or(..)
355            | CellRepr::Xor(..)
356            | CellRepr::Mux(..) => 1,
357            CellRepr::Adc(..) => 2,
358
359            CellRepr::Boxed(cell) => cell.output_len(),
360        }
361    }

363    pub(crate) fn visit(&self, mut f: impl FnMut(Net)) {
364        match self {
365            CellRepr::Void => unreachable!("void cell"),
366            CellRepr::Skip(_) => unreachable!("skip cell"),
367
368            CellRepr::Buf(arg) | CellRepr::Not(arg) => arg.visit(&mut f),
369            CellRepr::And(arg1, arg2) | CellRepr::Or(arg1, arg2) | CellRepr::Xor(arg1, arg2) => {
370                arg1.visit(&mut f);
371                arg2.visit(&mut f);
372            }
373            CellRepr::Mux(arg1, arg2, arg3) | CellRepr::Adc(arg1, arg2, arg3) => {
374                arg1.visit(&mut f);
375                arg2.visit(&mut f);
376                arg3.visit(&mut f);
377            }
378
379            CellRepr::Boxed(cell) => cell.visit(&mut f),
380        }
381    }

404impl Cell {
405    pub fn output_len(&self) -> usize {
406        match self {
407            Cell::Buf(arg) | Cell::Not(arg) => arg.len(),
408            Cell::And(arg1, arg2) | Cell::Or(arg1, arg2) | Cell::Xor(arg1, arg2) | Cell::Mux(_, arg1, arg2) => {
409                debug_assert_eq!(arg1.len(), arg2.len());
410                arg1.len()
411            }
412            Cell::Adc(arg1, arg2, _) => {
413                debug_assert_eq!(arg1.len(), arg2.len());
414                arg1.len() + 1
415            }
416
417            Cell::Eq(..) | Cell::ULt(..) | Cell::SLt(..) => 1,
418
419            Cell::Shl(arg1, _, _) | Cell::UShr(arg1, _, _) | Cell::SShr(arg1, _, _) | Cell::XShr(arg1, _, _) => {
420                arg1.len()
421            }
422
423            Cell::Mul(arg1, arg2)
424            | Cell::UDiv(arg1, arg2)
425            | Cell::UMod(arg1, arg2)
426            | Cell::SDivTrunc(arg1, arg2)
427            | Cell::SDivFloor(arg1, arg2)
428            | Cell::SModTrunc(arg1, arg2)
429            | Cell::SModFloor(arg1, arg2) => {
430                debug_assert_eq!(arg1.len(), arg2.len());
431                arg1.len()
432            }
433
434            Cell::Match(match_cell) => match_cell.output_len(),
435            Cell::Assign(assign_cell) => assign_cell.output_len(),
436
437            Cell::Dff(flip_flop) => flip_flop.output_len(),
438            Cell::Memory(memory) => memory.output_len(),
439            Cell::IoBuf(io_buffer) => io_buffer.output_len(),
440            Cell::Target(target_cell) => target_cell.output_len,
441            Cell::Other(instance) => instance.output_len(),
442
443            Cell::Input(_, width) => *width,
444            Cell::Output(..) => 0,
445            Cell::Name(..) | Cell::Debug(..) => 0,
446        }
447    }

449    pub fn has_effects(&self, design: &Design) -> bool {
450        match self {
451            Cell::IoBuf(_) | Cell::Other(_) | Cell::Input(..) | Cell::Output(..) | Cell::Name(..) => true,
452
453            Cell::Target(target_cell) => design.target_prototype(&target_cell).purity == TargetCellPurity::HasEffects,
454
455            _ => false,
456        }

459    pub fn visit(&self, mut f: impl FnMut(Net)) {
460        match self {
461            Cell::Input(..) => (),
462            Cell::Buf(arg) | Cell::Not(arg) | Cell::Output(_, arg) | Cell::Name(_, arg) | Cell::Debug(_, arg) => {
463                arg.visit(&mut f)
464            }
465            Cell::And(arg1, arg2)
466            | Cell::Or(arg1, arg2)
467            | Cell::Xor(arg1, arg2)
468            | Cell::Mul(arg1, arg2)
469            | Cell::UDiv(arg1, arg2)
470            | Cell::UMod(arg1, arg2)
471            | Cell::SDivTrunc(arg1, arg2)
472            | Cell::SDivFloor(arg1, arg2)
473            | Cell::SModTrunc(arg1, arg2)
474            | Cell::SModFloor(arg1, arg2)
475            | Cell::Eq(arg1, arg2)
476            | Cell::ULt(arg1, arg2)
477            | Cell::SLt(arg1, arg2)
478            | Cell::Shl(arg1, arg2, _)
479            | Cell::UShr(arg1, arg2, _)
480            | Cell::SShr(arg1, arg2, _)
481            | Cell::XShr(arg1, arg2, _) => {
482                arg1.visit(&mut f);
483                arg2.visit(&mut f);
484            }
485            Cell::Mux(net, value1, value2) | Cell::Adc(value1, value2, net) => {
486                value1.visit(&mut f);
487                value2.visit(&mut f);
488                net.visit(&mut f);
489            }
490            Cell::Match(match_cell) => match_cell.visit(&mut f),
491            Cell::Assign(assign_cell) => assign_cell.visit(&mut f),
492            Cell::Dff(flip_flop) => flip_flop.visit(&mut f),
493            Cell::Memory(memory) => memory.visit(&mut f),
494            Cell::IoBuf(io_buffer) => io_buffer.visit(&mut f),
495            Cell::Target(target_cell) => target_cell.visit(&mut f),
496            Cell::Other(instance) => instance.visit(&mut f),
497        }
498    }

521            | Cell::SShr(arg1, arg2, _)
522            | Cell::XShr(arg1, arg2, _) => {
